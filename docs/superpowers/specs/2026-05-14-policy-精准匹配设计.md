```typescript
interface PolicyCondition {
  // ── 基础工商信息 ──────────────────────────────
  registerCapital?: {
    min?: number;
    max?: number;
    unit?: '万' | '亿';
    currency?: '人民币' | '美元' | '欧元';
  };

  industry?: {
    required?: string[];      // 必备行业
    preferred?: string[];     // 优先行业
    excluded?: string[];      // 排除行业
  };

  region?: {
    province?: string;
    city?: string;
    district?: string;       // 如"浦东新区"
    taxLocation?: string;     // 纳税地要求
  };

  establishmentYear?: {
    min?: number;
    max?: number;
  };

  // ── 企业经营状况 ──────────────────────────────
  employeeCount?: {
    min?: number;
    max?: number;
  };

  annualRevenue?: {
    min?: number;
    max?: number;
    unit?: '万' | '亿';
  };

  taxAmount?: {
    min?: number;
    location?: string;
  };

  // ── 企业资质认证 ──────────────────────────────
  qualifications?: {
    required?: string[];     // 必备资质
    preferred?: string[];     // 优先有
  };

  creditRecord?: {
    required?: boolean;       // 是否要求无不良信用记录
  };

  // ── 研发创新相关 ──────────────────────────────
  rdExpenditure?: {
    minRatio?: number;        // 研发费用占营收比例 min
    minAmount?: number;       // 研发费用绝对值 min
  };

  intellectualProperty?: {
    patents?: { min?: number };
    softwareCopyright?: { min?: number };
  };

  // ── 特殊行业要求 ──────────────────────────────
  environmental?: {
    hasEIA?: boolean;         // 环评批复
    hasDischargePermit?: boolean;
  };

  safety?: {
    hasProductionLicense?: boolean;
  };

  importExport?: {
    hasImportExportRights?: boolean;
  };

  // ── 企业性质 ─────────────────────────────────
  enterpriseType?: {
    allowed?: string[];
    excluded?: string[];
    required?: string[];
  };

  listedStatus?: {
    allowed?: ('上市' | '非上市' | '新三板' | '科创板')[];
  };

  // ── 投资/用地 ────────────────────────────────
  projectInvestment?: {
    min?: number;
    unit?: '万' | '亿';
  };

  landArea?: {
    min?: number;
    buildingArea?: { min?: number; max?: number };
  };

  // ── 申报方式 ─────────────────────────────────
  applicationMethod?: string;  // 常年申报/定期申报

  // ── 元信息 ───────────────────────────────────
  specialConditions?: string[];
  rawText?: string;
  extractionStatus: 'pending' | 'completed' | 'failed';
  extractedAt?: string;
}
```

## 三、Supabase 表结构

### 3.1 policy_conditions 表

```sql
CREATE TABLE policy_conditions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  policy_id TEXT NOT NULL REFERENCES policies(id) ON DELETE CASCADE,
  conditions JSONB NOT NULL,
  raw_text TEXT,
  extraction_status TEXT DEFAULT 'pending' CHECK (extraction_status IN ('pending', 'completed', 'failed')),
  extraction_error TEXT,
  extracted_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(policy_id)
);

CREATE INDEX idx_policy_conditions_policy_id ON policy_conditions(policy_id);
CREATE INDEX idx_policy_conditions_status ON policy_conditions(extraction_status);
```

### 4.2 LLM Prompt 设计

```
## 任务
从政策申报条件文本中提取结构化条件。

## 输入
政策ID：{policyId}
政策名称：{policyName}
申报条件原文：
{policyConditionText}

## 输出要求
输出一个 JSON 对象，包含以下字段（没有的字段不要输出）：

- registerCapital: {min, max, unit, currency}
- industry: {required: string[], preferred: string[], excluded: string[]}
- region: {province, city, district, taxLocation}
- establishmentYear: {min, max}
- employeeCount: {min, max}
- annualRevenue: {min, max, unit}
- taxAmount: {min, location}
- qualifications: {required: string[], preferred: string[]}
- creditRecord: {required: boolean}
- rdExpenditure: {minRatio, minAmount}
- intellectualProperty: {patents: {min}, softwareCopyright: {min}}
- environmental: {hasEIA, hasDischargePermit}
- safety: {hasProductionLicense}
- importExport: {hasImportExportRights}
- enterpriseType: {allowed: string[], excluded: string[], required: string[]}
- listedStatus: {allowed: string[]}
- projectInvestment: {min, unit}
- landArea: {min, buildingArea: {min, max}}
- applicationMethod: string
- specialConditions: string[]

## 提取规则
1. 金额中的"以上""不低于""不少于"对应 min
2. 金额中的"以下""不超过""不多于"对应 max
3. 单位统一换算：万/亿
4. 行业名称标准化（如"集成电路"而非"IC"）
5. 只提取明确的条件，模糊表述不提取
6. 如果原文无某类条件，该字段不输出

## 输出格式
只输出 JSON，不要其他内容。
```

### 4.3 错误处理

```typescript
async function extractPolicyConditions(
  policyId: string,
  policyConditionText: string
): Promise<PolicyCondition> {
  try {
    const response = await callLLM({
      model: 'deepseek-chat',
      messages: [{ role: 'user', content: buildPrompt(policyId, policyConditionText) }],
      temperature: 0.1,  // 低温度保证稳定性
    });

    const conditions = JSON.parse(response.content);
    await saveToSupabase(policyId, conditions, 'completed');
    return conditions;
  } catch (error) {
    await saveToSupabase(policyId, null, 'failed', error.message);
    throw error;
  }
}
```

### 5.2 匹配算法

```typescript
interface MatchOutcome {
  matched: boolean;
  matchedConditions: string[];
  unmatchedConditions: string[];
  overallReason: string;
}

function matchEnterpriseToPolicy(
  enterprise: EnterpriseProfile,
  condition: PolicyCondition
): MatchOutcome {
  const matched: string[] = [];
  const unmatched: string[] = [];

  // 1. 注册资金
  if (condition.registerCapital) {
    const { min, max } = condition.registerCapital;
    const capital = enterprise.registeredCapital;
    if (min !== undefined && capital < min) {
      unmatched.push(`注册资金 ${capital}万 < ${min}万（要求）`);
    } else if (max !== undefined && capital > max) {
      unmatched.push(`注册资金 ${capital}万 > ${max}万（要求）`);
    } else {
      matched.push(`注册资金 ${capital}万 ${min !== undefined ? `≥ ${min}万` : ''}`);
    }
  }

  // 2. 行业
  if (condition.industry) {
    const entIndustries = enterprise.industry;
    if (condition.industry.required?.length) {
      const hasRequired = condition.industry.required.some(i =>
        entIndustries.some(ei => ei.includes(i) || i.includes(ei))
      );
      if (!hasRequired) {
        unmatched.push(`行业不匹配：要求 ${condition.industry.required.join('/')}，实际 ${entIndustries.join('/')}`);
      } else {
        matched.push(`行业匹配：${entIndustries.join('/')}`);
      }
    }
  }

  // 3. 区域
  if (condition.region) {
    const { district, taxLocation } = condition.region;
    if (district && !enterprise.district.includes(district)) {
      unmatched.push(`区域不匹配：要求 ${district}，实际 ${enterprise.district}`);
    }
    if (taxLocation && enterprise.taxLocation !== taxLocation) {
      unmatched.push(`纳税地要求：${taxLocation}`);
    }
  }

  // 4. 成立年限
  if (condition.establishmentYear) {
    const years = calculateYears(enterprise.establishmentDate);
    if (condition.establishmentYear.min && years < condition.establishmentYear.min) {
      unmatched.push(`成立年限不足：${years}年 < ${condition.establishmentYear.min}年`);
    }
    if (condition.establishmentYear.max && years > condition.establishmentYear.max) {
      unmatched.push(`成立年限超标：${years}年 > ${condition.establishmentYear.max}年`);
    }
  }

  // 5. 员工人数
  if (condition.employeeCount) {
    const { min, max } = condition.employeeCount;
    if (min !== undefined && enterprise.employeeCount < min) {
      unmatched.push(`员工人数不足：${enterprise.employeeCount}人 < ${min}人`);
    }
  }

  // 6. 资质认证
  if (condition.qualifications?.required?.length) {
    const hasAll = condition.qualifications.required.every(q =>
      enterprise.qualifications.some(eq => eq.includes(q))
    );
    if (!hasAll) {
      unmatched.push(`缺少必备资质：${condition.qualifications.required.join(', ')}`);
    } else {
      matched.push(`具备必备资质`);
    }
  }

  // 7. 信用记录
  if (condition.creditRecord?.required) {
    if (enterprise.creditRating === '不良') {
      unmatched.push(`信用记录不良`);
    }
  }

  // 8. 研发投入
  if (condition.rdExpenditure) {
    matched.push(`研发投入：需企业补充提供`);
  }

  // 9. 知识产权
  if (condition.intellectualProperty?.patels?.min) {
    if (enterprise.patents < condition.intellectualProperty.patents.min) {
      unmatched.push(`专利数量不足：${enterprise.patents} < ${condition.intellectualProperty.patents.min}`);
    } else {
      matched.push(`专利 ${enterprise.patents} 个达标`);
    }
  }

  return {
    matched: unmatched.length === 0,
    matchedConditions: matched,
    unmatchedConditions: unmatched,
    overallReason: unmatched.length === 0
      ? '符合所有申报条件'
      : `不符合 ${unmatched.length} 项条件`,
  };
}
```

## 八、风险与注意事项

1. **天眼查 API 限制**：500次/天，需控制调用频率
2. **LLM 提取准确性**：需人工抽查验证，可设置 extractionStatus 标记
3. **数据脱敏**：展示时隐藏法人手机号等敏感信息
4. **政策更新**：政策变更时需重新提取条件
